157 inline void add_systems(std::vector<System>& current_state, const std::vector<System>& new_state, double coef)
158 {
159 if (new_state.size() == 0) return;
160 size_t original_size = current_state.size();
161
162 current_state.insert(current_state.end(), new_state.begin(), new_state.end());
163
164 if (std::abs(coef - 1.0) > epsilon)
165 {
166 for (auto iter = current_state.begin() + original_size; iter != current_state.end(); ++iter)
167 {
168 iter->amplitude *= coef;
169 }
170 }
171
172 if (original_size != 0)
173 sort_merge_unique_erase(current_state, std::less<System>(),
174 std::equal_to<System>(), merge_system, remove_system);
175 }
