299 void generate_input(std::vector<System>& s, size_t addr_, size_t data_)
300 {
301 s.clear();
302
303 if (input_sz == pow2(addr_sz + data_sz))
304 {
305 generate_full_input(s, addr_, data_);
306 return;
307 }
308
309 unique_set.clear();
310 for (size_t i = 0; i < input_sz; )
311 {
312 auto&& input = rand_input();
313 auto&& res = unique_set.insert(input);
314 if (res.second)
315 {
316 i++;
317 s.emplace_back();
318 s.back().get(addr_).value = input.first;
319 s.back().get(data_).value = input.second;
320 }
321 }
322 SortUnconditional()(s);
323 Normalize()(s);
324 }

342 void generate_full_input(std::vector<System>& s, size_t addr_, size_t data_)
343 {
344 s.clear();
345 double amplitude = 1.0 / std::sqrt(input_sz);
346 for (size_t i = 0; i < pow2(addr_sz); ++i)
347 {
348 for (size_t j = 0; j < pow2(data_sz); ++j)
349 {
350 s.emplace_back();
351 s.back().get(addr_).value = i;
352 s.back().get(data_).value = j;
353 s.back().amplitude = amplitude;
354
355 unique_set.insert({ i,j });
356 }
