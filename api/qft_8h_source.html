160 inline void precompute_bitrev() {
161 profiler _("prefcompute_bitrev");
162 if (bitrev.empty())
163 {
164 bitrev.resize(full_size);
165#ifndef SINGLE_THREAD
166#pragma omp parallel for schedule(static)
167#endif
168 for (int64_t i = 0; i < full_size; ++i) {
169 uint64_t x = i;
170 uint64_t rev = 0;
171 for (size_t j = 0; j < n_digits; ++j) {
172 rev = (rev << 1) | (x & 1);
173 x >>= 1;
174 }
175 bitrev[i] = rev;
176 }
177 }
178 }
