276 {
277 size_t branch_sz = branches.size();
278 Branch::damp_prob_type ret;
279 if (branch_sz == 0) return ret;
280 ret.fill(0);
281 for (size_t bid = 0; bid < branch_sz; ++bid)
282 {
283 double prob = state_probs[bid];
284 auto&& damp_prob = branches[bid].get_prob_damp(qubit_id);
285 for (size_t i = 0; i < ret.size(); ++i)
286 {
287 ret[i] += damp_prob[i] * branch_probs[i];
