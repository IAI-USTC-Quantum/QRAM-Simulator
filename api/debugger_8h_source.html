369 template<typename OperationNq1q>
370 std::vector<size_t> check_inplace_unitarity(bool dagger)
371 {
372 System::clear();
373
374 size_t N = 2;
375 size_t reg1 = System::add_register("reg1", UnsignedInteger, N);
376 size_t reg2 = System::add_register("reg2", UnsignedInteger, 1);
377
378 OperationNq1q op("reg1", "reg2");
379
380 std::vector<bool> expected_output(pow2(N + 1), false);
381 std::vector<size_t> truth_table(pow2(N + 1), 0);
382
383 for (size_t i = 0; i < pow2(N); ++i)
384 {
385 for (size_t j = 0; j < pow2(1); ++j)
386 {
387 std::vector<System> state;
388 state.emplace_back();
389 state.back().get(reg1).value = i;
390 state.back().get(reg2).value = j;
391
392 if (dagger)
393 op.dag(state);
394 else
395 op(state);
396
397 if (state.size() != 1)
398 {
399 throw_bad_result("Bad check inplace unitarity: state size is not 1");
400 }
401
402 size_t out_i = state[0].get(reg1).value;
403 size_t out_j = state[0].get(reg2).value;
404 size_t out = (out_j << N) + out_i;
405 size_t in = (j << N) + i;
406
407 if (expected_output[out])
408 {
409 fmt::print("truth_table: {}", truth_table);
410 throw_bad_result("Bad check inplace unitarity: output state already exists");
411 }
412 expected_output[out] = true;
413 truth_table[in] = out;
414 }
415 }
416 System::clear();
417 return truth_table;
418 }

431 template<typename BlockEncoding, typename StateType = SparseState>
432 DenseMatrix<complex_t> _extract_block_encoding(BlockEncoding encA, std::string_view main_reg, std::string_view anc_UA,
433 bool is_full = false, bool is_dag = false)
434 {
435 size_t main_reg_num = System::size_of(main_reg);
436 size_t anc_UA_num = System::size_of(anc_UA);
437 size_t main_reg_pos = System::get(main_reg);
438 size_t anc_UA_pos = System::get(anc_UA);
439 if (is_full == true)
440 {
441 DenseMatrix<complex_t> ret(pow2(main_reg_num + anc_UA_num));
442 auto range_a = range(pow2(anc_UA_num));
443 auto range_m = range(pow2(main_reg_num));
444
445 for (auto [a, m] : product(range_a, range_m))
446 {
447 StateType state(1);
448
449 state.back().get(main_reg_pos).value = m;
450 state.back().get(anc_UA_pos).value = a;
451 if (is_dag)
452 encA.dag(state);
453 else
454 encA(state);
455 std::vector<complex_t> vec(pow2(main_reg_num + anc_UA_num), 0);
456 for (auto& s : state)
457 {
458 size_t _index = concat_value(
459 {
460 {s.get(main_reg_pos).value, main_reg_num},
461 {s.get(anc_UA_pos).value, anc_UA_num},
462 }
463 );
464 vec[_index] = s.amplitude;
465 }
466 size_t index = concat_value(
467 {
468 {m, main_reg_num},
469 {a, anc_UA_num},
470 }
471 );
472 for (size_t j = 0; j < pow2(main_reg_num + anc_UA_num); ++j)
473 {
474 ret(j, index) = vec[j];
475 }
476 }
477 return ret;
478 }
479 else
480 {
481 DenseMatrix<complex_t> ret(pow2(main_reg_num));
482
483 for (auto i : range(pow2(main_reg_num)))
484 {
485 StateType state(1);
486
487 state.back().get(main_reg_pos).value = i;
488 state.back().get(anc_UA_pos).value = 0;
489
490 if (is_dag)
491 encA.dag(state);
492 else
493 encA(state);
494
495 double prob = PartialTraceSelect({ {anc_UA, 0} })(state);
496
497 std::vector<complex_t> vec(pow2(main_reg_num), 0);
498 for (auto& s : state)
499 {
500 vec[s.get(main_reg_pos).value] = s.amplitude;
501 }
502 for (size_t j = 0; j < pow2(main_reg_num); ++j)
503 {
504 ret(j, i) = vec[j] / prob;
505 }
506 }
507 return ret;
508 }

534 inline void state_equal_check(std::vector<System> state1, std::vector<System> state2)
535 {
536 SortUnconditional()(state1);
537 SortUnconditional()(state2);
538 if (state1.size() != state2.size())
539 {
540 fmt::print("Size not equal: {} vs {}", state1.size(), state2.size());
541 goto CHECK_FAILED;
542 }
543 for (size_t i = 0; i < state1.size(); ++i)
544 {
545 if (state1[i] != state2[i])
546 {
547 fmt::print("State not equal at index {}:\n", i);
548 goto CHECK_FAILED;
549 }
550 }
551 return;
552 CHECK_FAILED:
553 fmt::print("State 1:\n");
554 StatePrint(0 | Detail)(state1);
555 fmt::print("State 2:\n");
556 StatePrint(0 | Detail)(state2);
557 throw_general_runtime_error("Hadamard_Bool: GPU-version failed!");
558 }
