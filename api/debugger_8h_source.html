378 template <typename Op>
379 std::vector<size_t> check_inplace_unitarity(
380 std::initializer_list<size_t> reg_sizes,
381 std::function<Op(std::vector<size_t>)> op_factory,
382 bool dagger = false)
383 {
384 System::clear();
385
386 std::vector<size_t> reg_ids;
387 std::vector<size_t> sizes_vec;
388 std::stringstream sbuf;
389 for (size_t i = 0; i < reg_sizes.size(); ++i) {
390 sbuf.str(""); sbuf << "_qram_tmp_" << i;
391 size_t sz = *(reg_sizes.begin() + i);
392 size_t id = System::add_register(sbuf.str(), UnsignedInteger, sz);
393 reg_ids.push_back(id);
394 sizes_vec.push_back(sz);
395 }
396
397 size_t total_bits = 0;
398 for (size_t sz : sizes_vec) total_bits += sz;
399 if (total_bits > 16)
400 fmt::print("WARNING: check_inplace_unitarity with {} total bits\n", total_bits);
401
402 Op op = op_factory(reg_ids);
403 const size_t N = pow2(total_bits);
404
405 std::vector<bool> output_seen(N, false);
406 std::vector<size_t> truth_table(N, 0);
407
408 for (size_t input = 0; input < N; ++input) {
409 std::vector<System> st;
410 st.emplace_back();
411
412 size_t remaining = input;
413 for (size_t ri = 0; ri < reg_ids.size(); ++ri) {
414 st[0].get(reg_ids[ri]).value = remaining & (pow2(sizes_vec[ri]) - 1);
415 remaining >>= sizes_vec[ri];
416 }
417
418 if (dagger) {
419 op.dag(st);
420 op(st);
421 } else {
422 op(st);
423 op.dag(st);
424 }
425
426 if (st.size() != 1)
427 throw_bad_result("check_inplace_unitarity: state split (non-unitary)");
428
429 size_t roundtrip = 0, shift = 0;
430 for (size_t ri = 0; ri < reg_ids.size(); ++ri) {
431 roundtrip |= (st[0].get(reg_ids[ri]).value & (pow2(sizes_vec[ri]) - 1)) << shift;
432 shift += sizes_vec[ri];
433 }
434 if (roundtrip != input)
435 throw_bad_result("check_inplace_unitarity: U*U† ≠ I");
436
437 size_t output = 0; shift = 0;
438 for (size_t ri = 0; ri < reg_ids.size(); ++ri) {
439 output |= (st[0].get(reg_ids[ri]).value & (pow2(sizes_vec[ri]) - 1)) << shift;
440 shift += sizes_vec[ri];
441 }
442 if (output_seen[output])
443 throw_bad_result("check_inplace_unitarity: non-bijective output collision");
444 output_seen[output] = true;
445 truth_table[input] = output;
446 }
447
448 System::clear();
449 return truth_table;
450 }

463 template<typename BlockEncoding, typename StateType = SparseState>
464 DenseMatrix<complex_t> _extract_block_encoding(BlockEncoding encA, std::string_view main_reg, std::string_view anc_UA,
465 bool is_full = false, bool is_dag = false)
466 {
467 size_t main_reg_num = System::size_of(main_reg);
468 size_t anc_UA_num = System::size_of(anc_UA);
469 size_t main_reg_pos = System::get(main_reg);
470 size_t anc_UA_pos = System::get(anc_UA);
471 if (is_full == true)
472 {
473 DenseMatrix<complex_t> ret(pow2(main_reg_num + anc_UA_num));
474 auto range_a = range(pow2(anc_UA_num));
475 auto range_m = range(pow2(main_reg_num));
476
477 for (auto [a, m] : product(range_a, range_m))
478 {
479 StateType state(1);
480
481 state.back().get(main_reg_pos).value = m;
482 state.back().get(anc_UA_pos).value = a;
483 if (is_dag)
484 encA.dag(state);
485 else
486 encA(state);
487 std::vector<complex_t> vec(pow2(main_reg_num + anc_UA_num), 0);
488 for (auto& s : state)
489 {
490 size_t _index = concat_value(
491 {
492 {s.get(main_reg_pos).value, main_reg_num},
493 {s.get(anc_UA_pos).value, anc_UA_num},
494 }
495 );
496 vec[_index] = s.amplitude;
497 }
498 size_t index = concat_value(
499 {
500 {m, main_reg_num},
501 {a, anc_UA_num},
502 }
503 );
504 for (size_t j = 0; j < pow2(main_reg_num + anc_UA_num); ++j)
505 {
506 ret(j, index) = vec[j];
507 }
508 }
509 return ret;
510 }
511 else
512 {
513 DenseMatrix<complex_t> ret(pow2(main_reg_num));
514
515 for (auto i : range(pow2(main_reg_num)))
516 {
517 StateType state(1);
518
519 state.back().get(main_reg_pos).value = i;
520 state.back().get(anc_UA_pos).value = 0;
521
522 if (is_dag)
523 encA.dag(state);
524 else
525 encA(state);
526
527 double prob = PartialTraceSelect({ {anc_UA, 0} })(state);
528
529 std::vector<complex_t> vec(pow2(main_reg_num), 0);
530 for (auto& s : state)
531 {
532 vec[s.get(main_reg_pos).value] = s.amplitude;
533 }
534 for (size_t j = 0; j < pow2(main_reg_num); ++j)
535 {
536 ret(j, i) = vec[j] / prob;
537 }
538 }
539 return ret;
540 }

566 inline void state_equal_check(std::vector<System> state1, std::vector<System> state2)
567 {
568 SortUnconditional()(state1);
569 SortUnconditional()(state2);
570 if (state1.size() != state2.size())
571 {
572 fmt::print("Size not equal: {} vs {}", state1.size(), state2.size());
573 goto CHECK_FAILED;
574 }
575 for (size_t i = 0; i < state1.size(); ++i)
576 {
577 if (state1[i] != state2[i])
578 {
579 fmt::print("State not equal at index {}:\n", i);
580 goto CHECK_FAILED;
581 }
582 }
583 return;
584 CHECK_FAILED:
585 fmt::print("State 1:\n");
586 StatePrint(0 | Detail)(state1);
587 fmt::print("State 2:\n");
588 StatePrint(0 | Detail)(state2);
589 throw_general_runtime_error("Hadamard_Bool: GPU-version failed!");
590 }
