85 struct StateStorage
86 {
88 uint64_t value = 0;
89
97 template<typename Ty>
98 Ty as(size_t size) const
99 {
100 uint64_t truncated_value = value & (pow2(size) - (size != 0));
101
102 if constexpr (std::is_floating_point_v<Ty>)
103 {
104 if (size == 64) return truncated_value * 1.0 / 2 / pow2(63);
105 return 1.0 * truncated_value / pow2(size);
106 }
107 else if constexpr (std::is_same_v<Ty, bool>)
108 {
109 return bool(truncated_value);
110 }
111 else if constexpr (std::is_integral_v<Ty>)
112 {
113 if constexpr (std::is_signed_v<Ty>)
114 {
115 return get_complement(truncated_value, size);
116 }
117 else
118 {
119 return truncated_value;
120 }
121 }
122 else {
123 throw_invalid_input();
124 }
125 }
