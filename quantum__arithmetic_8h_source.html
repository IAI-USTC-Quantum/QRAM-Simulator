1350 CustomArithmetic(const std::vector<size_t>& input_registers,
1351 size_t input_size, size_t output_size,
1352 GenericArithmetic func)
1353 : func(func)
1354 {
1355 if (input_ids.size() != output_ids.size())
1356 {
1357 throw std::invalid_argument("Input and output registers size does not match.");

1359 input_ids.resize(input_size);
1360 output_ids.resize(output_size);

1404 void operator()(std::vector<System>& state) const
1405 {
1406#ifdef SINGLE_THREAD
1407 for (auto& s : state)
1408 {
1409#else
1410#pragma omp parallel for
1411 for (int64_t i = 0; i < state.size(); ++i)
1412 {
1413 auto& s = state[i];
1414#endif
1415 if (ConditionNotSatisfied(s))
1416 continue;
1417
1418 /* obtain the value of the input registers */
1419 std::vector<size_t> input_values(input_ids.size());
1420 for (size_t i = 0; i < input_ids.size(); ++i)
1421 {
1422 input_values[i] = s.GetAs(input_ids[i], uint64_t);
1423 }
1424 /* apply the custom function */
1425 std::vector<size_t> output_values = func(input_values);
1426
1427 /* write the output reversibly via XOR */
1428 for (size_t i = 0; i < output_ids.size(); ++i)
1429 {
1430 s.get(output_ids[i]).value ^= output_values[i];
1431 }
1432 }
