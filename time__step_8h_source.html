185 template<typename BranchType>
186 inline void get_multiplier_qubit(
187 double gamma,
188 size_t step,
189 std::vector<BranchType>& branches,
190 size_t first_good_branch,
191 const std::vector<size_t>& good_branch_ids
192 ) const
193 {
194 ptrdiff_t addr_ref_count, addr_count;
195 addr_ref_count = _get_multiplier_impl_qubit(step,
196 branches[first_good_branch].address);
197
198 for (size_t i = 0; i < good_branch_ids.size(); ++i)
199 {
200 size_t id = good_branch_ids[i];
201 auto& branch = branches[id];
202
203 addr_count = _get_multiplier_impl_qubit(step,
204 branches[id].address);
205 branch.relative_multiplier = std::pow(1 - gamma, addr_count - addr_ref_count);
206 }
207 }
